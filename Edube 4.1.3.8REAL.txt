def is_year_leap(year):
	if year % 4 != 0:
		return False
	elif year % 100 != 0:
		return True
	elif year % 400 != 0:
		return False
	else:
		return True

def day_of_year(year, month, day):
    #Checks input
    if year < 1582:
        return None
    if month > 12 or month <1:
        return 1
    if day > days_in_month(year,month) or day < 1:# calls the function incase the moth doesnt have 31 days 
        return None 

    #calculates days
    totalDays = day# equals 31 
    month = month -1  # subtracts one from the month so if its feb then its will be jan
    while month > 0:
       totalDays += days_in_month(year,month)#adds the total days and finds the days of the month
       month -=1#subtracts one number from the month 
       
    return totalDays
